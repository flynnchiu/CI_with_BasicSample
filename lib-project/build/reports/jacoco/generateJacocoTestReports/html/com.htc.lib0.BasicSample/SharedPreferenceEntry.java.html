package com.htc.lib0.BasicSample;

import java.util.Calendar;

/**
 * Model class containing personal information that will be saved to SharedPreferences.
 */
public class SharedPreferenceEntry {

    // Name of the user.
    private final String mName;

    // Date of Birth of the user.
    private final Calendar mDateOfBirth;

    // Email address of the user.
    private final String mEmail;

    //private final String mPhone;
	
    public SharedPreferenceEntry(String name, Calendar dateOfBirth, String email) {
        mName = name;
        mDateOfBirth = dateOfBirth;
        mEmail = email;
    }
	/*	
    public SharedPreferenceEntry(String name, Calendar dateOfBirth, String email, String phone) {
        mName = name;
        mDateOfBirth = dateOfBirth;
        mEmail = email;
        mPhone = phone;
    }
		*/

    public String getName() {
        return mName;
    }

    public Calendar getDateOfBirth() {
        return mDateOfBirth;
    }

    public String getEmail() {
        return mEmail;
    }

}
